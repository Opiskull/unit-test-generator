using System;
using System.Collections;
using System.Linq;
using System.Text;
using Xunit;
using Moq;

namespace HelloWorldTest.Services
{
    public class HelloServiceTest
    {
        private readonly Mock<IBigMethod> _bigMethod = new Mock<IBigMethod>();
        private readonly Mock<IAnotherMethod> _anotherMethod = new Mock<IAnotherMethod>();
        private readonly HelloService _helloService;
        public HelloServiceTest()
        {
            _helloService = new HelloService(_bigMethod.Object, _anotherMethod.Object);
        }

        [Fact]
        public async Task ShouldTestBigIntMethod()
        {
            _bigMethod.Setup(_ => _.GetAll());
            await _helloService.BigIntMethod();
        }

        [Fact]
        public void ShouldTestBigMethod()
        {
            _helloService.BigMethod();
        }

        [Fact]
        public void ShouldTestAnotherMethod()
        {
            _bigMethod.Setup(_ => _.Bla());
            _helloService.AnotherMethod();
        }
    }
}
